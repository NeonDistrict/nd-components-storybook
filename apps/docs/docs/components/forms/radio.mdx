import { RadioGroup, RadioItem, Stack } from '@blockadegames/pizza-juice';

# Radio

Radios are used when only one choice may be selected in a series of options.

## Import

```jsx
import { RadioGroup, RadioItem } from '@blockadegames/pizza-juice';
```

## Usage

<RadioGroup defaultValue="1" label="Radio group">
  <Stack wrap="wrap">
    <RadioItem size="sm" value="1">
      Option 1
    </RadioItem>
    <RadioItem size="md" value="2">
      Option 2
    </RadioItem>
    <RadioItem size="lg" value="3">
      Option 3
    </RadioItem>
  </Stack>
</RadioGroup>
