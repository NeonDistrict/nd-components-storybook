import { useState } from 'react';
import { Stepper, Button, Stack } from '@blockadegames/pizza-juice';

# Stepper

Steppers display a list of high-level options that can expand/collapse to reveal more information.

## Import

```jsx
import { Stepper } from '@blockadegames/pizza-juice';
```

## Usage

export const CustomStepper = () => {
  const [step, setStep] = useState(1);
  const items = ['Item 1', 'Item 2', 'Item 3'];
  return (
    <>
      <Stepper items={items} activeItem={step} />
      <Stack justify="between" css={{ mt: '$4' }}>
        <Button onClick={() => setStep((e) => e - 1)} disabled={step === 1}>
          Prev Step
        </Button>
        <Button onClick={() => setStep((e) => e + 1)} disabled={step === 3}>
          Next Step
        </Button>
      </Stack>
    </>
  );
};

<CustomStepper />

```jsx
() => {
  const [step, setStep] = useState(1);
  const items = ['Item 1', 'Item 2', 'Item 3'];

  return (
    <>
      <Stepper items={items} activeItem={step} />

      <Stack justify="between" css={{ mt: '$4' }}>
        <Button onClick={() => setStep((e) => e - 1)} disabled={step === 1}>
          Prev Step
        </Button>

        <Button onClick={() => setStep((e) => e + 1)} disabled={step === 3}>
          Next Step
        </Button>
      </Stack>
    </>
  );
};
```

### Props

| Attribute  | Type       | Description         | Default |
| ---------- | ---------- | ------------------- | ------- |
| activeItem | `number`   | Current active item | `-`     |
| items      | `string[]` | List of items       | `[]`    |
