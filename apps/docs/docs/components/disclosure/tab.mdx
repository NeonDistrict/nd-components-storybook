# Tab

Tabs display a list of high-level options that can expand/collapse to reveal more information.

## Import

```jsx
import { Tab, TabList, TabItem } from '@blockadegames/pizza-juice';
```

## Usage

export const CustomTab = () => {
  return (
    <Tab>
      <TabList>
        <TabItem value="tab1">Tab 1</TabItem>
        <TabItem value="tab2">Tab 2</TabItem>
      </TabList>
    </Tab>
  );
};

<CustomTab />

---

```jsx
<Tab>
  <TabList>
    <TabItem value="tab1">Tab 1</TabItem>
    <TabItem value="tab2">Tab 2</TabItem>
  </TabList>
```
