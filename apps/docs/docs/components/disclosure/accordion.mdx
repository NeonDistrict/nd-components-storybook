import { Accordion, AccordionItem, Button } from '@blockadegames/pizza-juice';

# Accordion

Accordions display a list of high-level options that can expand/collapse to reveal more information.

## Import

```jsx
import { Accordion, AccordionItem } from '@blockadegames/pizza-juice';
```

## Usage

<Accordion>
  <AccordionItem title="Username" value="1">
    Username content
  </AccordionItem>

  <AccordionItem title="Email" value="2">
    Email content
  </AccordionItem>
</Accordion>

---

```jsx
<Accordion>
  <AccordionItem title="Username" value="1">
    Username content
  </AccordionItem>

  <AccordionItem title="Email" value="2">
    Email content
  </AccordionItem>
</Accordion>
```

## SubItems

<Accordion defaultValue="">
  <AccordionItem title="Username" value="1" subItem={<Button>SubItem</Button>}>
    Username content
  </AccordionItem>

  <AccordionItem title="Email" value="2" subItem={<Button>SubItem</Button>}>
    Email content
  </AccordionItem>
</Accordion>

---

```jsx
<Accordion defaultValue="">
  <AccordionItem title="Username" value="1" subItem={<Button>SubItem</Button>}>
    Username content
  </AccordionItem>

  <AccordionItem title="Email" value="2" subItem={<Button>SubItem</Button>}>
    Email content
  </AccordionItem>
</Accordion>
```
