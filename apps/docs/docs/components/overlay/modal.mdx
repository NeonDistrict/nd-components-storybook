import {
  Modal,
  ModalTitle,
  ModalDescription,
  Button,
  Box,
  Flex,
  useDisclosure,
} from '@blockadegames/pizza-juice';

# Modal

A modal is a dialog that focuses the user's attention exclusively on an information via a window that is overlaid on primary content.

## Import

```jsx
import { Modal, useDisclosure } from '@blockadegames/pizza-juice';
```

## Usage

export const ExampleModal = () => {
  const { isOpen, onOpen, onClose } = useDisclosure();
  return (
    <>
      <Button onClick={onOpen}>Open</Button>
      <Modal open={isOpen} onClose={onClose}>
        <Box css={{ color: '$white', bg: '$grey-900', w: 300, p: '$5' }}>
          <ModalTitle size="xl" weight="bold">
            Modal Title
          </ModalTitle>
          <ModalDescription css={{ mt: '$5' }}>
            Lorem, ipsum dolor sit amet consectetur adipisicing elit. Quo
            assumenda repellendus alias officia temporibus itaque rem, neque
            laudantium eligendi sapiente repudiandae quae, sequi vel modi culpa
            voluptatem quibusdam fugiat. Laboriosam.
          </ModalDescription>
          <Flex justify="end" css={{ mt: '$5' }}>
            <Button onClick={onClose}>Close</Button>
          </Flex>
        </Box>
      </Modal>
    </>
  );
};

<ExampleModal />

## Modal

### Props

| Attribute           | Type         | Description                                                                             | Default |
| ------------------- | ------------ | --------------------------------------------------------------------------------------- | ------- |
| open                | `boolean`    | The controlled open state of the dialog. Must be used in conjunction with onOpenChange. | `false` |
| portal              | `boolean`    | If `true` open modal on Portal                                                          | `true`  |
| closeOnOverlayClick | `boolean`    | Close modal when click outside the bounds of the component                              | `false` |
| closeOnEsc          | `boolean`    | Close modal when press `ESC`                                                            | `true`  |
| onClickOverlay      | `() => void` | Event handler called when a pointer event occurs outside the bounds of the component    | `-`     |
| onClose             | `() => void` | Event handler called when modal is closed                                               | `-`     |
| onEscapeKeyDown     | `() => void` | Event handler called when the escape key is down                                        | `-`     |
